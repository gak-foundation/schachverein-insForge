   ```ts
   });
   ```

3. **Fix `authorize()` in credentials provider** to actually verify:
   ```ts
   async authorize(credentials) {
     if (!credentials?.email || !credentials?.password) return null;
     const user = await db.query.users.findFirst({
       where: eq(users.email, credentials.email as string),
     });
     if (!user || !user.passwordHash) return null;
     const isValid = await verifyPassword(credentials.password as string, user.passwordHash);
     if (!isValid) return null;
     return { ...user, role: user.role ?? "mitglied", permissions: user.permissions ?? [], memberId: user.memberId };
   }
   ```

## Step 3: Auth Middleware

**New File:** `src/middleware.ts`

Protect dashboard routes server-side (currently only protected by `auth()` call in layout, but middleware provides earlier interception).

```ts
import { auth } from "@/lib/auth";
import { NextResponse } from "next/server";

export default auth((req) => {
  const { pathname } = req.nextUrl;
  const isAuth = !!req.auth;
  const isPublicRoute = pathname === "/" || pathname === "/login" || pathname === "/signup" || pathname.startsWith("/api/auth");

  if (!isAuth && !isPublicRoute) {
    return NextResponse.redirect(new URL("/login", req.url));
  }
  if (isAuth && (pathname === "/login" || pathname === "/signup")) {
    return NextResponse.redirect(new URL("/dashboard", req.url));
  }
  return NextResponse.next();
});

export const config = {
  matcher: ["/((?!_next/static|_next/image|favicon.ico|sw.js|manifest.json|icons/).*)"],
};
```

## Step 5: PWA — Service Worker

**New File:** `public/sw.js`

Cache-first strategy for static assets, network-first for API calls. This needs to be a plain JS file in `public/` (not compiled by Next.js).

```js
const CACHE_NAME = "schachverein-v1";
const STATIC_ASSETS = ["/", "/login", "/dashboard", "/manifest.json"];

self.addEventListener("install", (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(STATIC_ASSETS))
  );
  self.skipWaiting();
});

self.addEventListener("activate", (event) => {
  event.waitUntil(
    caches.keys().then((keys) =>
      Promise.all(keys.filter((k) => k !== CACHE_NAME).map((k) => caches.delete(k)))
    )
  );
  self.clients.claim();
});

self.addEventListener("fetch", (event) => {
  const { request } = event;
  if (request.method !== "GET") return;

  // API calls: network first, fall back to cache
  if (request.url.includes("/api/")) {
    event.respondWith(
      fetch(request).catch(() => caches.match(request))
    );
    return;
  }

  // Static assets: cache first, fall back to network
  event.respondWith(
    caches.match(request).then((cached) => cached || fetch(request).then((response) => {
      if (response.ok) {
        const clone = response.clone();
        caches.open(CACHE_NAME).then((cache) => cache.put(request, clone));
      }
      return response;
    }))
  );
});
```

**New File:** `src/components/pwa-register.tsx` — client component to register the service worker:
```tsx
"use client";
import { useEffect } from "react";

export function PwaRegister() {
  useEffect(() => {
    if ("serviceWorker" in navigator) {
      navigator.serviceWorker.register("/sw.js").catch(() => {});
    }
  }, []);
  return null;
}
```

## Step 6: Offline Sync — IndexedDB Utility

**New File:** `src/lib/offline/db.ts`

Foundation only — raw IndexedDB API (no extra dependency). Provides:

- `savePendingResult(result)` — store game results locally when offline

- `clearPendingResult(id)` — remove after successful sync
